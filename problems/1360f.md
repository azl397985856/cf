t = int(input())

# determin whether ans is at most one charactor diff from all elements of A
def can(m, n, ans, A):
    found = True
    for i in range(n):
        diff = 0
        # let's check
        for j in range(m):
            if ans[j] != A[i][j]:
                diff += 1
            if diff > 1:
                found = False
                break  # early return for a better performance
    return found


for _ in range(t):

    A = [input() for _ in range(n)]

        for j in range(26):
            ans = list(A[0])
            ans[i] = chr(ord("a") + j)
            found = can(m, n, ans, A)
            if found:  # early return for a better performance
                break
        if found:  # early return for a better performance

    if found:
